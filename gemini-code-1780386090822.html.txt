            width: 60px;

        }

<body>

    <div id="chessboard"></div>

    <script>
        const board = document.getElementById('chessboard');
        
        // Unicode chess pieces
        const pieces = {
            r: '♜', n: '♞', b: '♝', q: '♛', k: '♚', p: '♟', // Black
            R: '♖', N: '♘', B: '♗', Q: '♕', K: '♔', P: '♙'  // White
        };

        // Initial board setup (FEN-style string representation)
        const initialBoard = [
            'r', 'n', 'b', 'q', 'k', 'b', 'n', 'r',
            'p', 'p', 'p', 'p', 'p', 'p', 'p', 'p',
            '', '', '', '', '', '', '', '',
            '', '', '', '', '', '', '', '',
            '', '', '', '', '', '', '', '',
            '', '', '', '', '', '', '', '',
            'P', 'P', 'P', 'P', 'P', 'P', 'P', 'P',
            'R', 'N', 'B', 'Q', 'K', 'B', 'N', 'R'
        ];

        let draggedPiece = null;

        // Generate the board
        function createBoard() {
            for (let i = 0; i < 64; i++) {
                const square = document.createElement('div');
                square.classList.add('square');
                
                // Determine light or dark square
                const row = Math.floor(i / 8);
                const col = i % 8;
                if ((row + col) % 2 === 0) {
                    square.classList.add('light');
                } else {
                    square.classList.add('dark');
                }

                // Add piece if it exists in the initial setup
                const pieceSymbol = initialBoard[i];
                if (pieceSymbol) {
                    const piece = document.createElement('span');
                    piece.classList.add('piece');
                    piece.textContent = pieces[pieceSymbol];
                    piece.setAttribute('draggable', true);
                    
                    // Drag events for pieces
                    piece.addEventListener('dragstart', function(e) {
                        draggedPiece = piece;
                        setTimeout(() => piece.style.opacity = '0.5', 0);
                    });
                    
                    piece.addEventListener('dragend', function(e) {
                        setTimeout(() => piece.style.opacity = '1', 0);
                        draggedPiece = null;
                    });

                    square.appendChild(piece);
                }

                // Drop events for squares
                square.addEventListener('dragover', function(e) {
                    e.preventDefault(); // Necessary to allow dropping
                });

                square.addEventListener('drop', function(e) {
                    e.preventDefault();
                    if (draggedPiece) {
                        // If dropping on another piece, remove the captured piece
                        if (e.target.classList.contains('piece')) {
                            e.target.parentElement.appendChild(draggedPiece);
                            e.target.remove();
                        } else {
                            // If dropping on an empty square
                            this.appendChild(draggedPiece);
                        }
                    }
                });

                board.appendChild(square);
            }
        }

        createBoard();
    </script>
